#include<iostream>
#include<string.h>
#include<algorithm>
#include<climits>
#include<vector>
#include<map>
using namespace std;
const int maxN=1e3+5;
const int maxK=1e4+5;
char disk[maxN][maxK];
char toChar(int x){
    if(x<10)return (char)(x+'0');
    return (char)(x+'A'-10);
}
int toInt(char x){
    if(x<='9'&&x>='0')return (int)(x-'0');
    return (int)(x-'A'+10);
}
int main(){
    int n,s,l,x,y,len=0;
    scanf("%d %d %d",&n,&s,&l);
    for(int i=0;i<l;i++){
        scanf("%d ",&x);
        scanf("%s",disk[x]);
        len=(int)strlen(disk[x]);
    }
    int m;cin>>m;
    for(int h=0;h<m;h++){
        scanf("%d",&x);
        int row=x/(s*(n-1)*n)*n;
        int left=x%(s*(n-1)*n);
        int diskNum=left%(s*n)/s;
        int a=left/(s*n);
        if(diskNum+a>=n-1)a++;
        row+=a;
        //printf("%d %d\n",row,diskNum);
        y=8*(row*s+x%s);
        //cout<<y<<endl;
        if(y+8>len||n-l>1){
            printf("-\n");
            continue;
        }else if(y+8<=strlen(disk[diskNum])){
            for(int i=y;i<y+8;i++){
                printf("%c",disk[diskNum][i]);
            }printf("\n");
        }else{
            char ans[10]={"00000000"};
            for(int i=0;i<n;i++){
                if(i==diskNum)continue;
                for(int j=y;j<y+8;j++){
                    ans[j-y]=toChar(toInt(ans[j-y])^toInt(disk[i][j]));
                }
            }
            for(int i=0;i<8;i++){
                printf("%c",ans[i]);
            }printf("\n");
        }
    }
    return 0;
}
