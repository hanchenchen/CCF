# 201512-4-送货-[欧拉路径]-70

```c++
#include<bits/stdc++.h>
#include<iostream>
#include<vector>
#include<stack>
#include<math.h>
using namespace std;
const int maxN=1e4+5;
int n,m;
bool vis[maxN][maxN];
vector<int> g[maxN];
int main(){
    cin>>n>>m;
    int x,y;
    while(m--){
        cin>>x>>y;
        g[x].push_back(y);
        g[y].push_back(x);
    }
    int odd=(int)count_if(g+1,g+n+1,[](vector<int> v){return v.size()&1;});
    if(odd>2||(odd==2&&g[1].size()%2!=1)){
        cout<<"-1"<<endl;
        return 0;
    }
    for(int i=1;i<=n;i++){
        sort(g[i].begin(),g[i].end());
    }
    stack<int> st;
    vector<int> ans;
    st.push(1);
    while(!st.empty()){
        int x=st.top(),i;
        for(i=0;i<g[x].size();i++){
            if(!vis[x][g[x][i]]){
                vis[x][g[x][i]]=vis[g[x][i]][x]=true;
                st.push(g[x][i]);
                break;
            }
        }
        if(i==(int)g[x].size()){
            st.pop();
            ans.insert(ans.begin(),x);
        }
    }
    for(int i=0;i<ans.size();i++){
        if(i!=0)cout<<" ";
        cout<<ans[i];
    }
    return 0;   
}
```
