#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int n,arr[15][10];
int main(){
    int x=15,y=10;
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            cin>>arr[i][j];
        }
    }
    x=y=4;
    int num=0;
    int b[4][2],mid[4][4];
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            cin>>mid[i][j];
            if(mid[i][j]){
                b[num][0]=i;
                b[num][1]=j;num++;
            }
        }
    }
    cin>>x;
    x--;
    for(int i=0;i<4;i++){
        b[i][1]+=x;
    }
    while(b[0][0]<15){
        int flag=1;
        for(int i=0;i<4;i++){
            if(arr[b[i][0]+1][b[i][1]]==1||b[i][0]+1==15){
                flag=0;break;
            }
        }
        if(flag){
            for(int i=0;i<4;i++){
                b[i][0]++;
            }
        }else{
            break;
        }
    }
    for(int i=0;i<4;i++){
        arr[b[i][0]][b[i][1]]=1;
    }
    x=15;y=10;
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            if(j!=0)cout<<" ";
            cout<<arr[i][j];
        }
        cout<<endl;
    }
}
