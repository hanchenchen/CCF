```c++
#include<iostream>
#include<string.h>
#include<algorithm>
#include<climits>
using namespace std;
const int maxN=1e3+5;
const int maxK=1e3+5;
struct take{
    int id,start,minute;
}t[maxK];
int keyState[maxN]={0},box[maxN]={0},pos[maxN]={0};
int main(){
    int n,k;
    cin>>n>>k;
    for(int i=0;i<k;i++){
        scanf("%d %d %d",&t[i].id,&t[i].start,&t[i].minute);
    }
    sort(t,t+k,[](const take&a,const take&b){
        return a.start<b.start;
    });
    t[k].start=INT_MAX;
    int nowTime=0,left=0,i=0;
    while(left<n||i<k){
        nowTime=t[i].start;
        int backMin=INT_MAX;
        for(int j=1;j<=n;j++){
            if(keyState[j]<backMin){
                backMin=keyState[j];
            }
        }
        //cout<<nowTime<<" "<<backMin<<endl;
        if(nowTime<backMin){
            left--;
            keyState[t[i].id]=nowTime+t[i].minute;
            box[pos[t[i].id]]=0;
            pos[t[i].id]=0;
            /*for(int c=1;c<=n;c++){
                if(c!=1)printf(" ");
                printf("%d",box[c]);
            }printf("\n");*/
            i++;
        }else{
            int box_pos=1;
            for(int j=1;j<=n;j++){
                if(keyState[j]==backMin){
                    while(box[box_pos]!=0)box_pos++;
                    box[box_pos]=j;
                    pos[j]=box_pos;
                    keyState[j]=INT_MAX;
                    left++;
                }
            }
        }
    }
    /*int box_pos=1;
    for(int i=1;i<=n;i++){
        if(keyState[i]<INT_MAX){
            while(box[box_pos]!=0)box_pos++;
            box[box_pos]=i;
        }
    }*/
    for(int i=1;i<=n;i++){
        if(i!=1)printf(" ");
        printf("%d",box[i]);
    }
    return 0;
}
```
