#include<iostream>
using namespace std;
int n,arr[1000005]={0},x,y;
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>x>>y;
        arr[x]++;arr[y]--;
    }
    for(int i=0;i<n;i++){
        cin>>x>>y;
        arr[x]++;arr[y]--;
    }
    int ans=0;
    for(int i=1;i<=1000000;i++){
        arr[i]+=arr[i-1];
        if(arr[i]==2)ans++;
    }
    cout<<ans<<endl;
    return 0;
}
