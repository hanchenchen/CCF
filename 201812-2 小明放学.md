#include<iostream>
using namespace std;
int nextLight[4]={0,3,1,2},light[4];
void getTime(long long &x,long long &k,long long t){
    t%=(light[1]+light[2]+light[3]);
    while(t>0){
        if(t>=x){
            t-=x;
            k=nextLight[k];
            x=light[k];
            //cout<<t<<endl;
        }else{
            x-=t;
            break;
        }
    }
}
int main(){
    int n;
    cin>>light[1]>>light[2]>>light[3]>>n;
    long long k,t,ans=0;
    for(int i=0;i<n;i++){
        cin>>k>>t;
        if(k==0){
            ans+=t;continue;
        }
        getTime(t,k,ans);
        //cout<<k<<" "<<t<<endl;
        if(k==1)ans+=t;
        else if(k==2)ans+=t+light[1];
    }
    cout<<ans<<endl;
}
