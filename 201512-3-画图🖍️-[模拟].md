#include<bits/stdc++.h>
#include<iostream>
#include<vector>
#include<queue>
#include<math.h>
using namespace std;
const int maxN=1e2+5;
int n,m,q,movement[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
char arr[maxN][maxN];
struct p{int x,y;};
int main(){
    cin>>m>>n>>q;
    fill(arr[0],arr[0]
         +maxN*maxN,'.');
    int t,x1,y1,x2,y2;
    char c;
    while(q--){
        cin>>t;
        if(t){
            cin>>x1>>y1>>c;
            queue<p> q;
            q.push({x1,y1});
            while(!q.empty()){
                int x=q.front().x,y=q.front().y;
                q.pop();
                for(int i=0;i<4;i++){
                    if(y+movement[i][1]>=0&&y+movement[i][1]<n
                       &&x+movement[i][0]>=0&&x+movement[i][0]<m
                       &&arr[x+movement[i][0]][y+movement[i][1]]!='-'
                          &&arr[x+movement[i][0]][y+movement[i][1]]!='|'
                             &&arr[x+movement[i][0]][y+movement[i][1]]!='+'
                       &&arr[x+movement[i][0]][y+movement[i][1]]!=c
                          ){
                        arr[x+movement[i][0]][y+movement[i][1]]=c;
                        q.push({x+movement[i][0],y+movement[i][1]});
                    }
                }
            }
        }else{
            cin>>x1>>y1>>x2>>y2;
            if(x1==x2){
                for(int i=min(y1,y2);i<=max(y1,y2);i++){
                    if(arr[x1][i]=='-'||arr[x1][i]=='+')arr[x1][i]='+';
                    else arr[x1][i]='|';//'+' -> '+' 90->100
                }
            }else{
                for(int i=min(x1,x2);i<=max(x1,x2);i++){
                    if(arr[i][y1]=='|'||arr[i][y1]=='+')arr[i][y1]='+';
                    else arr[i][y1]='-';
                }
            }
        }
        
    }
    for(int j=n-1;j>=0;j--){
        for(int i=0;i<m;i++){
            cout<<arr[i][j];
        }
        cout<<endl;
    }
    return 0;
    
}
