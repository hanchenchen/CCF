```c++
//#include<bits/stdc++.h>
#include<iostream>
#include<vector>
#include<stack>
#include<algorithm>
#include<math.h>
using namespace std;
typedef long long ll;
const int maxN=1e2+5;
struct p{
    string lable="",id="";
    int level=0;
}html[maxN];
int n,m;
int main(){
    cin>>n>>m;
    getchar();
    string s;
    for(int i=0;i<n;i++){
        getline(cin,s);
        //cout<<s<<endl;
        int j=(int)count(s.begin(),s.end(),'.');
        html[i].level=j/2;
        int k=(int)s.find('#');
        if(k!=string::npos)
            html[i].id=s.substr(k);
        else
            k=(int)s.size()+1;
        html[i].lable=s.substr(j,k-j-1);
        //cout<< html[i].lable<<" "<<html[i].id<<" "<<html[i].level<<" "<<i<<" "<<n<<endl;
    }
    for(int i=0;i<m;i++){
        getline(cin,s);
        vector<string> tof;
        string mid;
        
        for(int j=0,k;j<s.size();j++){
            k=(int)s.find(' ',j);
            if(k==string::npos)k=(int)s.size();
            mid=s.substr(j,k-j);
            if(mid[0]!='#')transform(mid.begin(),mid.end(),mid.begin(),::tolower);
            tof.push_back(mid);
            j=k;
        }
        stack<int> st;vector<int> ans;
        for(int j=0;j<n;j++){
            while(!st.empty()&&html[j].level<=html[st.top()].level)st.pop();
            int len=(int)st.size();
            if(st.empty()||html[j].level>html[st.top()].level){
                if(tof[len]==html[j].lable||tof[len]==html[j].id){
                    st.push(j);
                }
            }
            if(st.size()==tof.size()){
                ans.push_back(j);
                st.pop();
            }
            
        }
        cout<<ans.size();
        for(int j=0;j<ans.size();j++){
            cout<<" "<<ans[j]+1;
        }
        cout<<endl;
    }
    return 0;
}
```
