# 201803-2 💥碰撞的小球

```c++
//#include<bits/stdc++.h>
#include<iostream>
#include<vector>
#include<queue>
#include<stdio.h>
using namespace std;
const int maxN=1e2+5;
const int maxL=1e3+5;
int main(){
    int n,l,t;
    cin>>n>>l>>t;
    int pos[maxN],vis[maxL]={0},v[maxN];
    for(int i=1;i<=n;i++){
        cin>>pos[i];
        v[i]=1;
    }
    while(t--){
        memset(vis,0,sizeof(vis));
        for(int i=1;i<=n;i++){
            pos[i]+=v[i];
            if(pos[i]==l||pos[i]==0)v[i]=-v[i];
            if(vis[pos[i]]){
                v[vis[pos[i]]]=-v[vis[pos[i]]];
                v[i]=-v[i];
            }else{
                vis[pos[i]]=i;
            }
        }
    }
    for(int i=1;i<=n;i++){
        if(i!=1)printf(" ");
        printf("%d",pos[i]);
    }
    return 0;
}
```
